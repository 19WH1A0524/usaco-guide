## Implementation

```cpp
#include <bits/stdc++.h>

using namespace std;

constexpr int nb_letter = 26;
using arr = array<int, nb_letter>;

void count_freq(const string &s, arr &freq) {
	for (char c: s) {
		freq[c - 'a']++;
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	freopen("blocks.in", "r", stdin);
	freopen("blocks.out", "w", stdout);
	int n;
	cin >> n;
	arr ans{};
	while (n--) {
		string s1, s2;
		cin >> s1 >> s2;
		arr freq1{}, freq2{};
		count_freq(s1, freq1); count_freq(s2, freq2);
		for (int i = 0; i < nb_letter; i++) {
			ans[i] += max(freq1[i], freq2[i]);
		}
	}
	for (int i: ans) {
		cout << i << '\n';
	}

	return 0;
}
```

```py
def countfreq(s):
	freqs = [0]*26
	for c in s:
		freqs[ord(c)-ord('a')] += 1
	return freqs

n = int(input())

ans = [0]*26
for i in range (n):
	s1, s2 = input().split()
	freq1, freq2 = countfreq(s1), countfreq(s2)

		ans[j] += max(freq1[j], freq2[j])
```

```java
import java.io.*;
import java.util.*;

public class BlockGame {

	/**
	 * Method counts every occurance of characters in a word.
	 * It returns the array containing letter frequency.
	 */
	private static int[] countChars(String word) {
		int[] freq = new int[26];
		for (int i = 0; i < word.length(); i++) {
			freq[word.charAt(i) - 'a']++;
		}
		return freq;
	}

	/**
	 * The method selects the maximum value for the frequency of individual letters in words,
	 * then adds the maximum to the sum of letter occurance in answer array.
	 */
	private static void addMaxFreq(int[] answer, int[] freq1, int[] freq2) {
		for (int i = 0; i < answer.length; i++) {
			answer[i] += Math.max(freq1[i], freq2[i]);
		}
	}

	public static void main(String[] args) throws IOException {
		Kattio io = new Kattio("blocks");
		int n = io.nextInt();
		int[] answer = new int[26];
		for (int i = 0; i < n; i++) {
			addMaxFreq(answer, countChars(io.next()), countChars(io.next()));
		}
		for (int i : answer) {
			io.println(i);
		}
		io.close();
	}
	
	//CodeSnip{Kattio}
}
```
