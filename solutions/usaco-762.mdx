```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
        freopen("homework.in", "r", stdin);
        int n;
        cin >> n;
        vector<ll> questions(n);
        for (int i = 0; i < n; i++) {
                cin >> questions[i];
        }
        int left = -1;
        int right = n;

        vector<ll> total_score(n);
        vector<ll> minimum_score(n);
        vector<ll> max_score_values;

        total_score[n - 1] = questions[n - 1];
        minimum_score[n - 1] = questions[n - 1];

        ll cur_q, cur_s, idx;
        for (int i = n - 2; i > 0; i--) {
                cur_q = questions[i];

                total_score[i] = cur_q + total_score[i + 1]; // Calculating sum of scores from index i to N
                minimum_score[i] = min(minimum_score[i + 1], cur_q); //minimum score from scores in indexes i to N

                cur_s = total_score[i] - minimum_score[i]; //excluding the least score question to calculate average
                idx = (n - (i + 1));

                if (cur_s * right > left * idx) {
                        left = cur_s;
                        right = idx;
                        max_score_values = {i};
                } else if (cur_s * right == left * idx) {
                        max_score_values.push_back(i);
                }
        }

        sort(max_score_values.begin(), max_score_values.end());

        freopen("homework.out", "w", stdout);
        for (ll k : max_score_values) {
                cout << k << "\n";
        }
}
```

with open('homework.in') as read:
        n = int(read.readline())
        questions = [int(i) for i in read.readline().split()]

left, right = -1, n
        
total = [0] * n
minimum = [0] * n
max_score_values = []

total[n - 1] = questions[n - 1]
minimum[n - 1] = questions[n - 1]

for i in range(n - 2, 0, -1):
        cur_q = questions[i]
        
        total[i] = cur_q + total[i + 1]
        minimum[i] = min(minimum[i + 1], cur_q)
        
        cur_s = total[i] - minimum[i]
        idx = (n - (i + 1))

        if cur_s * right > left * idx:
                left, right = cur_s, idx
                max_score_values = [i]

        elif cur_s * right == left * idx:
                max_score_values.append(i)

with open('homework.out', 'w') as out:
        for k in sorted(max_score_values):
                print(k, file=out)
