## Implementation

**Time Complexity:** $\mathcal{O}(N \log N)$

```cpp
CodeSnip{Benq Template}

int N, ans = 0;
map<int, int> mpp;
struct item {
    int q, t, x, n;
} p[200005];

bool cmp(const item& x, const item& y) {
    if (x.x - x.t == y.x - y.t)
        return x.q > y.q;

    return x.x - x.t < y.x - y.t;
}

int main() {
	setIO();
	re(N);
	F0R(i, N){
        re(p[i].q, p[i].t, p[i].x, p[i].n);
	}

    sort(p, p + N, cmp);

    F0R(i, N) {
        if (p[i].q == 2) {
            mpp[p[i].x + p[i].t] += p[i].n;
        }
        else {
            int n = p[i].n;

            while (n) {
                map<int, int>::iterator it = mpp.lower_bound(p[i].x + p[i].t);

                if (it == mpp.end()) break;
                int u = min(n, it->second);

                if (u == it->second) {
                    mpp.erase(it);
                }
                else {
                    it->second -= u;
                }

                n -= u;
                ans += u;
            }
        }
    }

    ps(ans);
}
```
