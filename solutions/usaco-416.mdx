<CPPSection>

```cpp
#include <iostream>
#include <fstream>
#include <vector>

using std::cout;
using std::endl;
using std::vector;

int main() {
	std::ifstream read("lazy.in");

	int n, k;
	read >> n >> k;

	/*
	 * the 45 degree rotation expands the size of the field
	 * it also includes the cells with half distance
	 */
	vector<vector<int>> field(2 * n + 1, vector<int>(2 * n + 1));

	// read input and store it rotated by 45 degrees
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			read >> field[i + j][n - i + j - 1];
		}
	}

	/*
	 * used to avoid non-lattice points &
	 * ensures that the dummy 0's don't affect anything
	 */
	int t = (n + 1) % 2;

	n = n * 2 - 1;
	int best = 0;
	for (int i = 0; i < n; i++) {
		/*
		 *  you can go at most k steps, but we do 2k because we set 
		 *  zeros in between which doubles the distance
		 *  also, Bessie can only be positioned in lattice points
		 */
		int sum = 0;
		for (int a = std::max(i - k, 0); a < n && a <= i + k; a++) {
			for (int b = std::max(t - k, 0); b < n && b <= t + k; b++) {
				sum += field[a][b];
			}
		}

		best = std::max(best, sum);

		// slide the region
		for (int j = t + 1; j + k < n; j++) {
			for (int a = std::max(i - k, 0); a < n && a <= i + k; a++) {
				if (j - k - 1 >= 0) {
					sum -= field[a][j - k - 1];
				}
				sum += field[a][j + k];
			}
			// update the sum only in lattice points
			if (j % 2 == t && best < sum) {
				best = sum;
			}
		}

		t = 1 - t;
	}

	std::ofstream("lazy.out") << best << endl;
}
```
